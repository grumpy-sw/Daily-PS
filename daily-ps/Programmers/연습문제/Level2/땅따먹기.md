func solution(_ land:[[Int]]) -> Int{
    var answer = 0
    let N = land.count
    let C = 4
    var dp = Array(repeating: Array(repeating: 0, count: 4), count: N)
        
    for i in 0..<C {
        dp[0][i] = land[0][i]
    }
        
    for i in 1..<N {
        for j in 0..<C {
            for k in 0..<C {
                if j == k {
                    continue
                }
                dp[i][j] = max(dp[i][j], dp[i-1][k] + land[i][j])
            }
        }
    }

    return dp[N-1].max()!
}
