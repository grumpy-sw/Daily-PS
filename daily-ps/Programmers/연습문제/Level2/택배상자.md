func solution(_ order:[Int]) -> Int {
    let N = order.count
    var index = 0
    var stack: [Int] = []
    var count = 0
    
    for box in 1...N {
        if box != order[index] {
            stack.append(box)
        } else {
            index += 1
            count += 1
            
            while !stack.isEmpty && stack.last! == order[index] {
                index += 1
                count += 1
                stack.removeLast()
            }
        }
    }
    
    return count
}
