func solution(_ line:[[Int]]) -> [String] {
    var dots: [[Int]] = []
        
    var maxX = Int.min
    var minX = Int.max
    var maxY = Int.min
    var minY = Int.max
        
    for i in 0..<line.count-1 {
        let A = line[i][0]
        let B = line[i][1]
        let E = line[i][2]
            
        for j in (i+1)..<line.count {
            let C = line[j][0]
            let D = line[j][1]
            let F = line[j][2]
                
            if A*D == B*C {
                continue
            }
                
            let x = (B*F - E*D)/(A*D - B*C)
            let y = (E*C - A*F)/(B*C - A*D)
                
            if Double(B*F - E*D)/Double(A*D - B*C) == Double(x), Double(E*C - A*F)/Double(B*C - A*D) == Double(y) {
                dots.append([x, y])
                    
                if minX > x {
                    minX = x
                }
                if maxX < x {
                    maxX = x
                }
                if minY > y {
                    minY = y
                }
                if maxY < y {
                    maxY = y
                }
            }
        }
    }
        
    let R = maxY - minY + 1
    let C = maxX - minX + 1
        
    var map: [[String]] = Array(repeating: Array(repeating: ".", count: C), count: R)
        
    for dot in dots {
        map[dot[1] - minY][dot[0] - minX] = "*"
    }
        
    return map.map{ $0.joined() }
}
